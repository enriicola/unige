case CONFIGWIZARD_MODE_GETSTAGE2HTML:

  $hostname = grab_array_var($inargs, 'hostname', gethostbyaddr($address));

  $rp_port = nagiosccm_replace_user_macros($port);
  $rp_token = nagiosccm_replace_user_macros($token);
  $services_serial = grab_array_var($inargs, "services_serial", "");
  if ($services_serial) {
    $services = unserialize(base64_decode($services_serial));
  }
  $not_used = array();
  $return_code = 0;
  $alternative_host_check = false;
  exec('ping -W 2 -c 1 ' . $rp_address, $not_used, $return_code);

Question 11 Answer
a.

the $no_ssl_verify variable can be misuses for command injection purposes.

The code is not vulnerable.

c.

d.

The $address variable can be misused for command injection purposes.
